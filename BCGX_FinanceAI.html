#Graphs comparing the companies over the last 3 years

sns.set(style="whitegrid")

# Define the list of financial metrics and their respective titles
metrics = ['Total Revenue', 'Net Income', 'Cash Flow from Operating Activities', 'Total Assets', 'Total Liabilities ']
titles = ['Total Revenue (in millions)', 'Net Income (in millions)', 'Cash Flow from Operating Activities (in millions)', 
          'Total Assets (in millions)', 'Total Liabilities (in millions)']

# Plots for every single metric
for i, metric in enumerate(metrics):
    plt.figure(figsize=(10, 6))
    
    # Plot data for each company separately
    for company in df['Company'].unique():
        subset = df[df['Company'] == company]
        plt.plot(subset['Year'], subset[metric], marker='o', label=company)
    
    # Plot styling
    plt.title(f'{titles[i]} Over Time')
    plt.xlabel('Year')
    plt.ylabel(titles[i])
    plt.legend(title='Company')
    plt.xticks(df['Year'].unique())
    plt.grid(axis='y', linestyle='--', alpha=0.7)
    
    # Show the plot
    plt.show()

Microsoft: Revenue grew by 6.88% in 23 followed by a 15.67% increase in 24, small decrease in net income in 23 (-0.52%) followed by a massive increase of 21.80% in 24, cash flow declined slightly in 23 (-1.6%) before seeing an increase in 24 (35.36%)

Tesla: Massive growth of 18.79% in 23 followed by a growth of 0.95% in 24, net income dropped by over 50% in 24 (-52.23%), cash flow saw a decrease in 23 but rebounded pretty well in 24.
